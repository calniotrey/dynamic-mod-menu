dmm_button_effect = {
	potential = {
		always = yes
	}
	allow = {
		# always = yes
		hidden_trigger = {
			NOT = {
				has_country_flag = dmm_opened
			}
		}
	}
	effect = {
		custom_tooltip = dmm_button_effect_tooltip
		hidden_effect = {
			save_event_target_as = dmm_country_caller
			set_variable = {
				which = dmm_starting_leader_index
				value = 0
			}
			set_country_flag = dmm_opened
			country_event = {
				id = dmm_events.0
			}
		}
	}
}
